#include<iostream>
using namespace std;

int main()
{
    int n;
    cout << "Enter the size of array: ";
    cin >> n;
    cout << "Enter the array elements:\n";
    int arr[n];
    for(int i = 0; i < n; i++)
    {
        cin >> arr[i];
    }
    int x;
    cout << "Enter the difference to find in keys: ";
    cin >> x;
    for( int i = 0; i< n;i ++)
    {
        for( int j = 0; j < n; j++)
        {
            if( arr[j] - arr[i] == x)
            {
                cout << "\nThe value representation is: ";
                    cout << arr[j] << " - " << arr[i] << " = " << x << endl;
                    
                    cout << "The index representation is: ";
                    cout << j << " , "<< i << " , "<< x << "\n" << endl;
            }
        }
    }
}
